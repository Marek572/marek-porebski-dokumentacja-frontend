import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { Observable } from 'rxjs';
import { IErrorAPIModel } from 'src/app/models/errorAPI.model';
import { ISchoolModel } from 'src/app/models/school/school.model';
import { IUserModel } from 'src/app/models/user/user.model';
import { InterviewsService } from 'src/app/services/interviews/interviews.service';
import { SchoolsService } from 'src/app/services/schools/schools.service';
import { UsersService } from 'src/app/services/users/users.service';
import { InterviewUserFormComponent } from '../interview-user-form/interview-user-form.component';
import { DataLossDialogComponent } from '../../dialog/dataLoss-dialog/dataLoss-dialog/dataLoss-dialog.component';

@Component({
  selector: 'app-interview-form',
  templateUrl: './interview-form.component.html',
  styleUrls: ['./interview-form.component.scss'],
})
export class InterviewFormComponent implements OnInit {

  constructor(
    private formBuilder: FormBuilder,
    public dialog: MatDialog,
    public schoolService: SchoolsService,
    public interviewService: InterviewsService,
    private snackBar: MatSnackBar
  ) { }

  interviewForm!: FormGroup;
  schoolList: ISchoolModel[] = [];
  selectedDate: Date | null = new Date;
  displayedColumns: string[] = ['name', 'surname', 'actions'];
  interviewUsers: IUserModel[] = [];

  errorAPI: IErrorAPIModel = { type: '' };

  ngOnInit() {

    this.schoolService.getSchools().subscribe({
      next: (schools: ISchoolModel[]) => {
        this.schoolList = schools
      }
    })

    this.interviewForm = this.formBuilder.group({
      schoolId: ['', Validators.required],
      date: [this.selectedDate],
      users: [''],
      termsNumber: ['', Validators.compose([
        Validators.required,
        Validators.min(0),
      ])],
      classNumber: ['', Validators.compose([
        Validators.required,
        Validators.min(0),
        Validators.max(8),
      ])]
    }, { updateOn: 'blur' })

  }

  onDateChange(newDate: Date) {
    this.interviewForm.get('date')?.setValue(newDate);
  }

  refreshTable() {
    this.interviewUsers = [...this.interviewUsers];
  }

  addBtnClick() : void {
    console.log('add user to interview');
    const dialogRef = this.dialog.open(InterviewUserFormComponent, {
      data: this.interviewUsers,
      disableClose: true
    });

    dialogRef.componentInstance.userAdded.subscribe((user: IUserModel) => {
      this.interviewUsers.push(user);
      this.refreshTable();
      dialogRef.close();
    });
  }

  deleteBtnClick(interviewUser: IUserModel) : void {
    console.log('delete ', interviewUser);
    const startFrom = this.interviewUsers.indexOf(interviewUser)
    this.interviewUsers.splice(startFrom, 1)
    this.refreshTable();
  }

  onFormSubmit() : void {
    
    console.log(this.interviewUsers);
    if (this.interviewForm.invalid)
      return;
    
    else {


      const userIds = this.interviewUsers.map(user => user.id)
      this.interviewForm.get('users')?.setValue(userIds)

      console.log(this.interviewForm.value);
      this.interviewService.addInterview(this.interviewForm.value)
        .subscribe({
          next: () => {},
          error: (err: IErrorAPIModel)=>{
            console.log(err);
            // handleError(this.userForm,err);
          },
          complete: ()=>{
            this.snackBar.open('Wywiadówka dodana pomyślnie', '', {
              duration: 4000,
              panelClass: ['custom-snackbar']
            });
            this.dialog.closeAll();
          }
        })
    }
  }

  onFormCancel() : void {
    if (!this.interviewForm.dirty&&this.interviewUsers.length==0) {
      this.dialog.closeAll();
    }else{
      let dialogRef = this.dialog.open(DataLossDialogComponent, {});
    }
  }

}
