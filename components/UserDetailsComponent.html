import { Component, Inject, OnInit } from '@angular/core';
import { MAT_DIALOG_DATA, MatDialog } from '@angular/material/dialog';
import { ChartDataset, ChartOptions } from 'chart.js/auto';
import { IStatsModel, IUserStatsModel } from 'src/app/models/stats/stats.model';

@Component({
  selector: 'app-details',
  templateUrl: './user-details.component.html',
  styleUrls: ['./user-details.component.scss']
})

export class UserDetailsComponent implements OnInit {

  constructor(
    @Inject(MAT_DIALOG_DATA) public data: IUserStatsModel,
    public dialog: MatDialog,
  ) { }

  dataKeys: Array<string> = [];

  chartOptions: ChartOptions = {
    color: 'white',
    responsive: true,
    scales: {
      x: {
        title: {
          display: true,
          text: 'Data wywiadówki',
          color: 'white',
          font: {
            size: 15,
            weight: 'bold'
          }
        },
        ticks: {
          maxRotation: 0,
          minRotation: 0,
          color: 'white'
        },
        grid: {
          color: 'white'
        }
      },
      y: {
        title: {
          display: true,
          text: 'Fiszki',
          color: 'white',
          font: {
            size: 15,
            weight: 'bold'
          }
        },
        beginAtZero: true,
        ticks: {
          precision: 0,
          color: 'white'
        },
        grid: {
          color: 'white'
        }
      },
    },
    plugins: {
      colors: {
        enabled: false
      },
    }
  };

  chartLabels: string[] = this.data.stats.map((stat: IStatsModel) => new Date(stat.date).toLocaleDateString('default', { day: '2-digit', month: '2-digit' }));
  chartData: ChartDataset[] = [
    {
      data: this.data.stats.map((stat: IStatsModel) => stat.count),
      label: 'Zebrane fiszki',
      backgroundColor: 'orange'
    }
  ];

  ngOnInit() {

    console.log(this.data)
    this.dataKeys = Object.keys(this.data.user)

  }

  closeBtnClick() {
    this.dialog.closeAll();
  }



}
