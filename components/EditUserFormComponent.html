import { Component, Inject, OnInit } from '@angular/core';
import { MAT_DIALOG_DATA, MatDialog } from '@angular/material/dialog';
import { IErrorAPIModel } from 'src/app/models/errorAPI.model';
import { UsersService } from 'src/app/services/users/users.service';
import { MatchData, handleError } from 'src/app/utils/common-functions';
import { MatSnackBar } from '@angular/material/snack-bar';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { IUpdateUserModel } from 'src/app/models/user/update-user.model';
import { Job, Role } from 'src/app/models/user/enum.model';
import { DataLossDialogComponent } from '../../dialog/dataLoss-dialog/dataLoss-dialog/dataLoss-dialog.component';

@Component({
  selector: 'app-edit-user-form',
  templateUrl: './edit-user-form.component.html',
  styleUrls: ['./edit-user-form.component.scss'],
})

export class EditUserFormComponent implements OnInit {

  constructor(private formBuilder: FormBuilder,
    public dialog: MatDialog,
    private snackBar: MatSnackBar,
    public userService: UsersService,
    @Inject(MAT_DIALOG_DATA) public editUserData: IUpdateUserModel
  ) { }
  userEditForm!: FormGroup;
  roleList = Role;
  jobList = Job;



  ngOnInit() {

    const { name, surname, phone, email, role, job, active } = this.editUserData

    this.userEditForm = this.formBuilder.group({
      name: [name, Validators.compose([
        Validators.required,
        Validators.minLength(3),
        Validators.maxLength(30)
      ])],
      surname: [surname, Validators.compose([
        Validators.required,
        Validators.minLength(3),
        Validators.maxLength(30)
      ])],
      phone: [phone, Validators.compose([
        Validators.required,
        Validators.pattern(new RegExp('^(?=.*[0-9]).{9,9}$'))
      ])],
      email: [email, Validators.compose([
        Validators.required,
        Validators.email
      ])],
      role: [Object.keys(this.roleList).find(e => e === role), Validators.required],
      job: [Object.keys(this.jobList).find(e => e === job), Validators.required],
      active: [active, Validators.required],
    }, { updateOn: 'blur' })
  }

  getDirtyData() {
    const dirtyData: any = {};

    Object.keys(this.userEditForm.controls).forEach(key => {

      const control = this.userEditForm.get(key);

      if (control?.dirty)
        dirtyData[key] = control.value;

    });

    return dirtyData;
  }



  onFormEditSubmit() {

    if (this.userEditForm.invalid)
      return

    if (this.userEditForm.dirty) {
      console.log(this.userEditForm.value)
      this.userService.updateUser(this.getDirtyData(), this.editUserData.id)
        .subscribe({
          next: () => { },
          error: (err: IErrorAPIModel) => {
            handleError(this.userEditForm, err)
          },
          complete: () => {
            this.dialog.closeAll();
            this.snackBar.open('Użytkownik zmieniony pomyślnie', '', {
              duration: 4000,
              panelClass: ['custom-snackbar']
            });
          }
        })
    }
  }

  onFormCancel() {
  
    if (MatchData(this.userEditForm.value,this.editUserData,['name', 'surname', 'phone', 'email', 'role', 'job'])) {
      this.dialog.closeAll();
    }else{
      let dialogRef = this.dialog.open(DataLossDialogComponent, {});
    }
    
  }
}
