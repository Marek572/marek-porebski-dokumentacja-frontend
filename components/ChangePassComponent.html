import { Component, Inject, OnInit } from '@angular/core';
import { FormBuilder, FormControl, FormGroup, Validators } from '@angular/forms';
import { MatDialogRef } from '@angular/material/dialog';
import { AuthService } from 'src/app/services/auth/auth.service';
import { Router } from '@angular/router';
import { IChangePassModel } from 'src/app/models/user/change-pass.model';


@Component({
  selector: 'app-ChangePass',
  templateUrl: './ChangePass.component.html',
  styleUrls: ['./ChangePass.component.scss']
})
export class ChangePassComponent implements OnInit {

  constructor(
    public dialogRef: MatDialogRef<ChangePassComponent>,
    private formBuilder: FormBuilder,
    private authService:AuthService,
    @Inject(Router) private router: Router
  ) { }
  newPassword = new FormControl('');
  hideOld: boolean=true;
  hideNew!: boolean;
  repeatNewPassword!: boolean;
  userForm!: FormGroup;
  showDetails: boolean = false;

  ngOnInit(): void {
    this.userForm = this.formBuilder.group({
      OldPassword: ['', Validators.compose([
        Validators.required,
        Validators.minLength(6),
        Validators.maxLength(30)
      ])],
      NewPassword: ['', Validators.compose([
        Validators.required,
        Validators.minLength(6),
        Validators.maxLength(30)
      ])],
      RepeatNewPassword: ['', Validators.compose([
        Validators.required,
        Validators.minLength(6),
        Validators.maxLength(30),
      ]),]
    }, {updateOn: 'blur' })
  }
  async onSubmit(changePassForm: IChangePassModel) {
    const ControlRepeatNewPassword = this.userForm.get('RepeatNewPassword');
    if(this.userForm.get('NewPassword')?.value===this.userForm.get('RepeatNewPassword')?.value){
      console.log(changePassForm)
      this.authService.changePassword(changePassForm).subscribe((loggedIn) => {
        if (loggedIn) {
          this.dialogRef.close();
          this.authService.logout();
          this.router.navigate(['/'])
        }
      })
    }else{
      ControlRepeatNewPassword?.setErrors({ notSame: true });
    }

    
  }
  onNoClick(): void {
    localStorage.setItem("AskChangePassword", "No")
    this.dialogRef.close();
  }

  getPasswordErrorMessageOldPassword() {
    const ControlOldPassword = this.userForm.get('OldPassword');
    
    if (ControlOldPassword?.hasError('required')) {
      return 'Hasło jest wymagane.';
    } 

    return '';
  }
  getPasswordErrorMessageNewPassword() {
    const ControlNewPassword = this.userForm.get('NewPassword');
    if (ControlNewPassword?.hasError('required')) {
      return 'Hasło jest wymagane.';
    } else if (ControlNewPassword?.hasError('minlength')) {
      return 'Hasło musi być dłuższe niż 5 znaki';
    } else if (ControlNewPassword?.hasError('maxlength')) {
      return 'Hasło musi być krótsze niż 30 znaków';
    }
   
    return '';
  }
  getPasswordErrorMessageRepeatNewPassword() {
    const ControlRepeatNewPassword = this.userForm.get('RepeatNewPassword');
    if (ControlRepeatNewPassword?.hasError('required')) {
      return 'Hasło jest wymagane.';
    } else if (ControlRepeatNewPassword?.hasError('minlength')) {
      return 'Hasło musi być dłuższe niż 5 znaki';
    } else if (ControlRepeatNewPassword?.hasError('maxlength')) {
      return 'Hasło musi być krótsze niż 30 znaków';
    }else if (this.userForm.get('NewPassword')?.value!=this.userForm.get('RepeatNewPassword')?.value) {
      return 'Hasła nie są takie same.';
    }

    return '';
  }
  onYesClick(): void {
    this.dialogRef.close(true);
  }


}
