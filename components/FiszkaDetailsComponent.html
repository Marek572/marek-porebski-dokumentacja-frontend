import { Component, Inject, OnInit } from '@angular/core';
import { FormBuilder, FormGroup } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialog, MatDialogRef } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { IFiszkaModel } from 'src/app/models/fiszka/fiszka.model';
import { IUpdatePhoneCallsModel } from 'src/app/models/phone-calls/update-phone-calls.model';
import { FiszkaService } from 'src/app/services/fiszka/fiszka.service';
import { PhoneCallFormComponent } from '../../forms/phone-call-form/phone-call-form.component';
import { PhoneCallService } from 'src/app/services/phone-call/phone-call.service';
import { IPhoneCallsModel } from 'src/app/models/phone-calls/phone-calls.model';

@Component({
  selector: 'app-fiszka-details',
  templateUrl: './fiszka-details.component.html',
  styleUrls: ['./fiszka-details.component.scss']
})
export class FiszkaDetailsComponent implements OnInit {

  constructor(
    @Inject(MAT_DIALOG_DATA) public fiszka: IFiszkaModel,
    public dialog: MatDialog,
    public dialogRef: MatDialogRef<FiszkaDetailsComponent>,
    private formBuilder: FormBuilder,
    private fiszkaService: FiszkaService,
    public phoneCallService: PhoneCallService,
    private snackBar: MatSnackBar
  ) { }

  isPhoneCallEdited: boolean = false;
  resignationField!: FormGroup;
  fiszkaCommentsForm!: FormGroup;

  ngOnInit(): void {

    console.log(this.fiszka);

    this.resignationField = this.formBuilder.group({
      resignation: [this.fiszka.resignation]
    }, {updateOn: 'blur'})

    this.fiszkaCommentsForm = this.formBuilder.group({
      comments: [this.fiszka.comments]
    }, {updateOn: 'blur'})

  }

  closeBtnClick(): void {

    this.dialogRef.close();

  }

  resignationSwitch(value: boolean): void {

    const resignationValue = { id: this.fiszka.id, resignation: value }

    this.fiszkaService.updateFiszka(resignationValue, this.fiszka.id).subscribe({
      next: () => {},
      error: (err: any) => {console.log(err);},
      complete: ()=>{
        this.snackBar.open('Pole rezygnacja zaktualizowane', '', {
          duration: 4000,
          panelClass: ['custom-snackbar']
        });
      }
    });

  }

  //FIXME: z jakiegos powodu nie odswieza danych dialogu po dodaniu rozmowy, w innych przypadkach dziala?
  addPhoneCall(): void {

    console.log('addPhoneCall');

    this.dialog.open(PhoneCallFormComponent, {
      data: this.fiszka.id,
      disableClose: true
    });

  }

  // editPhoneCall(phoneCall: IUpdatePhoneCallsModel): void {

  //   console.log(phoneCall);

  // }

  deletePhoneCall(phoneCall: IPhoneCallsModel): void {

    console.log(phoneCall);

    this.phoneCallService.deletePhoneCall(phoneCall).subscribe({
      next: () => { },
      error: (err) => {console.log(err);},
      complete: () => {
        this.snackBar.open('Osoba kontaktowa usunięta', '', {
          duration: 4000,
          panelClass: ['custom-snackbar']
        });
        //FIXME: z jakiegos powodu nie odswieza danych dialogu po usunieciu rozmowy, w innych przypadkach dziala?
        this.fiszkaService.fiszkiHasChanged.next(true);
      }
    });

  }

  updateFiszkaComments(): void {

    if (this.fiszkaCommentsForm.dirty && this.fiszkaCommentsForm.valid) {
      console.log(this.fiszkaCommentsForm.value);
      this.fiszkaService.updateFiszka(this.fiszkaCommentsForm.value, this.fiszka.id).subscribe({
        next: () => {},
        error: (err: any) => {console.log(err);},
        complete: ()=>{
          this.snackBar.open('Komentarze zaktualizowane', '', {
            duration: 4000,
            panelClass: ['custom-snackbar']
          });
        }
      });
    }

  }

}
