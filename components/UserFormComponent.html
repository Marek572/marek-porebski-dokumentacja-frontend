import { Component, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { IErrorAPIModel } from 'src/app/models/errorAPI.model';
import { UsersService } from 'src/app/services/users/users.service';
import { PhoneValidator } from 'src/app/validators/phoneValidator'
import { EmailValidator } from 'src/app/validators/emailValidator'
import { MatSnackBar } from '@angular/material/snack-bar';
import {  handleError } from 'src/app/utils/common-functions';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Job, Role } from 'src/app/models/user/enum.model'
import { DataLossDialogComponent } from '../../dialog/dataLoss-dialog/dataLoss-dialog/dataLoss-dialog.component';

@Component({
  selector: 'app-form',
  templateUrl: './user-form.component.html',
  styleUrls: ['./user-form.component.scss']
})

export class UserFormComponent implements OnInit {

  constructor(
    private formBuilder: FormBuilder,
    public dialog: MatDialog,
    public userService: UsersService,
    public phoneValidator: PhoneValidator,
    public emailValidator: EmailValidator,
    private snackBar: MatSnackBar
  ) { }

  userForm!: FormGroup;
  isActiveUser = true;
  errorAPI: IErrorAPIModel = { type: '' };
  jobList = Job;
  roleList = Role;

  ngOnInit() {

    this.userForm = this.formBuilder.group({
      name: ['', Validators.compose([
        Validators.required,
        Validators.minLength(3),
        Validators.maxLength(30)
      ])],
      surname: ['', Validators.compose([
        Validators.required,
        Validators.minLength(3),
        Validators.maxLength(30)
      ])],
      phone: ['', Validators.compose([
        Validators.required,
        Validators.pattern(new RegExp('^(?=.*[0-9]).{9,9}$'))
      ]), [this.phoneValidator]],
      email: ['', Validators.compose([
        Validators.required,
        Validators.email
      ]), [this.emailValidator]],
      role: ['', Validators.required],
      job: ['', Validators.required],
      active: [this.isActiveUser, Validators.required],
    }, { updateOn: 'blur' })

  }

  onFormSubmit() {

    if (this.userForm.invalid)
      return;

    else {
      this.userService.addUser(this.userForm.value)
        .subscribe({
          next: () => {},
          error: (err: IErrorAPIModel)=>{
            handleError(this.userForm,err);
          },
          complete: ()=>{
            this.snackBar.open('Użytkownik dodany pomyślnie', '', {
              duration: 4000,
              panelClass: ['custom-snackbar']
            });
            this.dialog.closeAll();
          }
        })
    }
  }

  onFormCancel() {
    if (!this.userForm.dirty) {
      this.dialog.closeAll();
    }else{
      let dialogRef = this.dialog.open(DataLossDialogComponent, {});
    }
  }
}
