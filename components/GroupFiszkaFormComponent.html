import { Component, Inject, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialog, MatDialogRef } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { IFiszkaModel } from 'src/app/models/fiszka/fiszka.model';
import { IGroupModel } from 'src/app/models/group/group.model';
import { IInterviewsModel } from 'src/app/models/interviews/interviews.model';
import { FiszkaService } from 'src/app/services/fiszka/fiszka.service';
import { InterviewsService } from 'src/app/services/interviews/interviews.service';

@Component({
  selector: 'app-group-fiszka-form',
  templateUrl: './group-fiszka-form.component.html',
  styleUrls: ['./group-fiszka-form.component.scss']
})
export class GroupFiszkaFormComponent implements OnInit {

  constructor(
    @Inject(MAT_DIALOG_DATA) public data: IGroupModel,
    private formBuilder: FormBuilder,
    public dialog: MatDialog,
    public dialogRef: MatDialogRef<GroupFiszkaFormComponent>,
    public interviewsService: InterviewsService,
    public fiszkaService: FiszkaService,
    private snackBar: MatSnackBar
  ) { }

  groupFiszkaForm!: FormGroup;

  isInterviewSelected: boolean = false;
  interviewsList: IInterviewsModel[] = [];
  fiszkiInInterview: IFiszkaModel[] = [];

  ngOnInit() {

    this.interviewsService.getInterviews().subscribe({
      next: (interviews: IInterviewsModel[]) => {
        this.interviewsList = interviews
      },
      error: (err) => {console.log(err);},
    })

    this.groupFiszkaForm = this.formBuilder.group({
      fiszka: ['', Validators.required],
    });

  }

  onInterviewSelectChange(interviewId: string): void {

    this.isInterviewSelected = true;

    this.fiszkaService.getFiszkiByInterviewId(interviewId).subscribe({
      next: (fiszki: IFiszkaModel[]) => {
        let alreadyAddedFiszki = fiszki.filter(fiszka => !this.data.fiszki.find(e => e.id == fiszka.id))
        this.fiszkiInInterview = alreadyAddedFiszki
      },
      error: (err) => {console.log(err);},
    })

  }

  onFormSubmit(): void {

    console.log(this.groupFiszkaForm.value.fiszka);

    if (this.groupFiszkaForm.invalid)
      return
    else {
      this.fiszkaService.addFiszkaToGroup(this.groupFiszkaForm.value, this.data.id).subscribe({
        next: () => {},
        error: (err) => {console.log(err);},
        complete: () => {
          this.snackBar.open('Fiszka dodana pomyślnie', '', {
            duration: 4000,
            panelClass: ['custom-snackbar']
          });
          this.dialogRef.close();
        }
      })
    }

  }

  onFormCancel(): void {
    this.dialogRef.close();
  }

}

<form [formGroup]="groupFiszkaForm" class="formStyle">

  <mat-form-field>
    <mat-select placeholder="Wywiadówka" (selectionChange)="onInterviewSelectChange($event.value.id)">

        {{ interview.school.name }} {{ interview.date | date: 'dd/MM/yyyy' }}
      </mat-option>

  </mat-form-field>

        {{ fiszka.name }} {{ fiszka.surname }}
      </mat-option>
    </mat-select>
    <mat-error>

        Fiszka jest wymagana.
      }
    </mat-error>
  </mat-form-field>

  <div class="actionButtonsContainer" matDialogActions>
    <button mat-button>

    </button>
    <button mat-button>
      <div class="btn submitBtn" (click)="onFormSubmit()">Dodaj</div>
    </button>
  </div>

</form>
