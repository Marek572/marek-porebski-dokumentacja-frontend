import { Component, EventEmitter, Inject, OnInit, Output } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialog, MatDialogRef } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { IUserModel } from 'src/app/models/user/user.model';
import { UsersService } from 'src/app/services/users/users.service';

@Component({
  selector: 'app-interview-user-form',
  templateUrl: './interview-user-form.component.html',
  styleUrls: ['./interview-user-form.component.scss']
})
export class InterviewUserFormComponent implements OnInit {

  constructor(
    @Inject(MAT_DIALOG_DATA) public data: IUserModel[],
    private formBuilder: FormBuilder,
    public dialog: MatDialog,
    public dialogRef: MatDialogRef<InterviewUserFormComponent>,
    public userService: UsersService,
    private snackBar: MatSnackBar
  ) { }

  interviewUserForm!: FormGroup;
  usersInRegion: IUserModel[] = [];

  @Output() userAdded = new EventEmitter<IUserModel>();

  ngOnInit() {

    this.userService.getUsers().subscribe({
      next: (users: IUserModel[]) => {
        this.usersInRegion = users
      },
      error: (err) => {console.log(err);},
      complete: () => {
        this.usersInRegion = this.usersInRegion.filter(userInRegion => !this.data.some(interviewUser => interviewUser.id === userInRegion.id));
      }
    })

    this.interviewUserForm = this.formBuilder.group({
      user: ['', Validators.required],
    });

  }

  onFormSubmit(): void {

    console.log(this.interviewUserForm.value.user);
    

    if (this.interviewUserForm.invalid)
      return
    else {
      this.userAdded.emit(this.interviewUserForm.value.user);
      this.snackBar.open('Użytkownik dodany pomyślnie', '', {
        duration: 4000,
        panelClass: ['custom-snackbar']
      });
    }

  }

  onFormCancel(): void {
    this.dialogRef.close();
  }

}
