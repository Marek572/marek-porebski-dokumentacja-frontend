import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { AuthService } from '../../services/auth/auth.service';
import { MatDialog } from '@angular/material/dialog';

@Component({
  selector: 'app-navbar',
  templateUrl: `./navbar.component.html`,
  styleUrls: ['./navbar.component.scss']
})

export class NavbarComponent implements OnInit {

  constructor(
    private router: Router,
    private authService: AuthService,
    private dialog: MatDialog,
  ) {  }

  isMenuOpen: boolean = false;

  onMenuToggle(): void {
    this.isMenuOpen = !this.isMenuOpen;
  }

  onMenuMouseLeave(): void {
    if (this.isMenuOpen) {
      this.isMenuOpen = false;
    }
  }

  currentUserRole = this.authService.loggedUser.value?.role;

  loggedUser: string | null = '';
  loggedUserName: string | null = '';
  loggedUserSurname: string | null = '';
  firstTime: boolean | null = null;

  profile = 'Profil';
  users = 'Pracownicy';
  fiszki = 'Fiszki';
  interviews = 'Wywiadówki';
  groups = 'Grupy';
  stats = 'Statystyki';
  facilities = 'Placówki';

  profileIcon = 'person';
  usersIcon = 'group';
  fiszkiIcon = 'sentiment_satisfied';
  interviewsIcon = 'checklist';
  groupsIcon = 'groups'
  statsIcon = 'leaderboard';
  facilitiesIcon = 'apartment';

  profileLink = '/profile';
  usersLink = '/users';
  fiszkiLink = '/fiszki';
  interviewsLink = '/interviews';
  groupsLink = '/groups'
  statsLink = '/stats';
  facilitiesLink = '/schools'

  // profileLink = `/${this.currentUserRole}/profile`;
  // usersLink = `/${this.currentUserRole}/users`;
  // fiszkiLink = `/${this.currentUserRole}/fiszki`;
  // interviewsLink = `/${this.currentUserRole}/interviews`;
  // groupsLink = `/${this.currentUserRole}/groups`;
  // statsLink = `/${this.currentUserRole}/stats`;
  // facilitiesLink = `/${this.currentUserRole}/schools`;

  ngOnInit(): void {

    //FIXME: przy reloadzie strony nie przypisuje wartosci
    //console.log(this.globalService.customVariable)
    //this.loggedUser = this.globalService.customVariable;
    // this.loggedUser = this.auth.loggedUser;
    this.loggedUser = localStorage.getItem('account');
    this.loggedUserName = localStorage.getItem('account');

  }
  onProfileClick() {
    this.router.navigate(['/profile'])
  }

  onLogoutClick() {
    this.authService.logout();
  }

}
