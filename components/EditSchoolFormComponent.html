import { Component, Inject, OnInit } from '@angular/core';
import { MAT_DIALOG_DATA, MatDialog } from '@angular/material/dialog';
import { IErrorAPIModel } from 'src/app/models/errorAPI.model';
import { SchoolsService } from 'src/app/services/schools/schools.service';
import { IUpdateSchoolModel } from 'src/app/models/school/update-school.model';
import { STEPPER_GLOBAL_OPTIONS } from '@angular/cdk/stepper';
import { MatchData, handleError } from 'src/app/utils/common-functions';
import { MatSnackBar } from '@angular/material/snack-bar';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Province, Status, Type } from 'src/app/models/school/enum.model';
import { DataLossDialogComponent } from '../../dialog/dataLoss-dialog/dataLoss-dialog/dataLoss-dialog.component';

@Component({
  selector: 'app-school-form',
  templateUrl: './edit-school-form.component.html',
  styleUrls: ['./edit-school-form.component.scss'],
  providers: [
    {
      provide: STEPPER_GLOBAL_OPTIONS,
      useValue: { displayDefaultIndicatorType: false },
    },
  ],
})

export class EditSchoolFormComponent implements OnInit {

  constructor(
    private formBuilder: FormBuilder,
    public dialog: MatDialog,
    public schoolService: SchoolsService,
    private snackBar: MatSnackBar,
    @Inject(MAT_DIALOG_DATA) public editSchoolData: IUpdateSchoolModel
  ) { }


  currentTabIndex: number = 0;
  maxTabIndex: number = 2;
  statusList = Status;
  typeList = Type;
  provinceList = Province;
  schoolDetailsForm!: FormGroup;
  schoolAdressForm!: FormGroup;
  schoolContactForm!: FormGroup;
  errorAPI: IErrorAPIModel = { type: '' };

  ngOnInit() {

    const { name, status, type, cooperation, street, suiteNumber, postCode, city, province, phone, email } = this.editSchoolData

    this.schoolDetailsForm = this.formBuilder.group({
      name: [name, Validators.required],
      status: [Object.keys(this.statusList).find(e => e === status), Validators.required],
      type: [Object.keys(this.typeList).find(e => e === type), Validators.required],
      cooperation: [cooperation],
    }, { updateOn: 'blur' })

    this.schoolAdressForm = this.formBuilder.group({
      street: [street, Validators.required],
      suiteNumber: [suiteNumber, Validators.required],
      postCode: [postCode, Validators.compose([
        Validators.required,
        Validators.pattern(new RegExp('^\\d{2}-\\d{3}$'))
      ])],
      city: [city, Validators.required],
      province: [Object.keys(this.provinceList).find(e => e === province), Validators.required],
    }, { updateOn: 'blur' })

    this.schoolContactForm = this.formBuilder.group({
      phone: [phone, Validators.compose([
        Validators.required,
        Validators.pattern(new RegExp('^(?=.*[0-9]).{9,9}$'))
      ])],
      email: [email, Validators.compose([
        Validators.email
      ])],
    }, { updateOn: 'blur' })

  }

  checkValidators(): boolean {
    const schoolDetailsFormValid = this.schoolDetailsForm.valid;
    const schoolAdressValid = this.schoolAdressForm.valid;
    const schoolContactValid = this.schoolContactForm.valid;

    if (schoolDetailsFormValid && schoolAdressValid && schoolContactValid)
      return true
    else
      return false;
  }

  getDirtyData() {
    let combinedDirtyValues: any = {};

    const formGroups = [
      this.schoolDetailsForm,
      this.schoolAdressForm,
      this.schoolContactForm
    ];

    formGroups.forEach(formGroup => {
      Object.keys(formGroup.controls).forEach(key => {
        const currentControl = formGroup.controls[key];
        if (currentControl.dirty) {
          combinedDirtyValues[key] = currentControl.value;
        }
      });
    });

    return combinedDirtyValues;
  }

  onFormSubmit() {

    if (!this.checkValidators())
      return;

    if (this.schoolDetailsForm.dirty || this.schoolAdressForm.dirty || this.schoolContactForm.dirty) {
      console.log('addSchool');
      this.schoolService.updateSchool(this.getDirtyData(), this.editSchoolData.id)
        .subscribe({
          next: () => { },
          error: (err: IErrorAPIModel) => {
            handleError(this.schoolContactForm, err);
          },
          complete: () => {
            this.dialog.closeAll();
            this.snackBar.open('Szkoła zmieniona pomyślnie', '', {
              duration: 4000,
              panelClass: ['custom-snackbar']
            });
          }
        })
    }
  }

  onFormCancel() {
    const merged = Object.assign(this.schoolDetailsForm.value, this.schoolAdressForm.value, this.schoolContactForm.value);
    const matcher = ["name", "status", "type", "cooperation", "street", "suiteNumber", "city", "province", "phone", "email"]
    if (MatchData(merged, this.editSchoolData, matcher)) {
      this.dialog.closeAll();
    } else {
      let dialogRef = this.dialog.open(DataLossDialogComponent, {});
    }
  }

}
