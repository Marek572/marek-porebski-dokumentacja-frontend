import { Injectable } from '@angular/core'
import { HttpEvent, HttpInterceptor, HttpHandler, HttpRequest, HttpErrorResponse, HttpResponse } from '@angular/common/http'
import { Observable, tap, catchError, filter, of, map, iif, first, skip, switchMap, throwError } from 'rxjs'
import { AuthService } from 'src/app/services/auth/auth.service'

@Injectable()
export class TokenInterceptor implements HttpInterceptor {
  noTokenUrls = ['http://localhost:3000/auth/login']

  constructor(private readonly auth: AuthService) {}

  refreshTokenUrl = 'http://localhost:3000/auth/refreshToken'
  refreshTokenInProgress = false

  intercept(request: HttpRequest<any>, next: HttpHandler): Observable<HttpEvent<any>> {
    // 1. dodaj token do request-a
    // 2. zrob next.handle(nowyRequest)
    // 3. sprawdz odpowiedz - jesli 200 tok, jesli 401 to zrob (a) refresh (b) ponownie strzał z requestem

    return this.sendRequest(request, next)
      .pipe(
        catchError((httpError: HttpErrorResponse, obs) =>
          httpError?.status === 401 && !this.noTokenUrls.some(el => el === request.url)
            // token wygasł, trzeba zrobić refresh
            ? this.handleTokenRefresh(request, next)
            : throwError(() => httpError.error)
        )
      )
  }

  private sendRequest(request: HttpRequest<any>, next: HttpHandler) {
    return this.waitForNewTokenIfNeeded(request)
      .pipe(
        this.addTokenToRequestIfNeeded(request),
        switchMap(requestWithToken => next.handle(requestWithToken)),
        // filtrujemy event HttpSentEvent ({type: 0}) - type = 0 oznacza że zapytanie zostało wysłane, czekamy na HttpResponse
        filter(res => res instanceof HttpResponse)
      )
  }
  private waitForNewTokenIfNeeded(request: HttpRequest<any>) {
    const shouldWaitForNewToken = () => this.refreshTokenInProgress && request.url !== this.refreshTokenUrl
    return iif(
      shouldWaitForNewToken,
      this.waitForNewToken(),
      of(null)
    )
  }

  private addTokenToRequestIfNeeded(request: HttpRequest<any>) {
    const newRequest =
      this.noTokenUrls.some(el => el === request.url)
        ? of(request)
        : this.auth.getTokenData()
          .pipe(
            first(),
            map(access_token => {
              if (!access_token) {
                console.log('addtokentorequest', access_token);
                return request
              } else {
                console.log('bearer',access_token);
                let headers = request.headers.set('Authorization', `Bearer ${access_token}`)
                // klon, nie można ponownie użyć zmienionego HttpRequest
                return request.clone({ headers })
              }
            })
          )
    return switchMap(() => newRequest)
  }

  private handleTokenRefresh(request: HttpRequest<any>, next: HttpHandler) {
    return this.postTokenRefresh()
      .pipe(
        this.addTokenToRequestIfNeeded(request),
        switchMap(next.handle)
      )
  }
  private postTokenRefresh() {
    return of(null).pipe(
      tap(() => this.refreshTokenInProgress = true),
      switchMap(() => this.auth.refreshJWT()),
      tap(() => this.refreshTokenInProgress = false),
    )
  }
  private waitForNewToken() {
    return this.auth.getTokenData().pipe(
      skip(1),
      first()
    )
  }
}
