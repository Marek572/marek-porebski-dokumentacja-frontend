export class UserDetails implements IShowDetails {

    private userService: UsersService;
    private dialog: MatDialog;

    constructor(private injector: Injector) {
        this.userService = this.injector.get(UsersService);
        this.dialog = this.injector.get(MatDialog);
    }

    showDetails(userId: string): void {

        console.log('user ' + userId);

        //FIXME: daty testowe, do zmiany na koniec
        // let startDate = new Date();
        let startDate = new Date('2023-05-01T06:35:22.000Z');
        let endDate = new Date();
        // endDate.setMonth(startDate.getMonth() - 1);

        console.log(startDate, endDate);

        combineLatest([
            this.userService.getUserById(userId),
            this.userService.getUserStatsInRange( startDate, endDate ,userId),
        ]).subscribe(([userData, userStats]) => {
            this.dialog.open(UserDetailsComponent, {
                data: {user: userData, stats: userStats},
                disableClose: true
            });
        });

    }

}

export class SchoolDetails implements IShowDetails {

    private schoolService: SchoolsService;
    private directorService: DirectorService;
    private rentalCostService: RentalCostService;
    private contactPersonService: ContactPersonService;
    private dialog: MatDialog;
    private dialogRef: MatDialogRef<SchoolDetailsComponent> | null = null;

    constructor(private injector: Injector) {
        this.schoolService = this.injector.get(SchoolsService);
        this.directorService = this.injector.get(DirectorService);
        this.rentalCostService = this.injector.get(RentalCostService);
        this.contactPersonService = this.injector.get(ContactPersonService);
        this.dialog = this.injector.get(MatDialog);
    }


    showDetails(schoolId: string): void {

        console.log('school ' + schoolId);
        let schoolDetailsData: ISchoolModel;

        combineLatest([
            this.schoolService.getSchoolsById(schoolId),
            this.directorService.getDirector(schoolId),
            this.rentalCostService.getRentalCost(schoolId),
            this.contactPersonService.contactPersonHasChanged.pipe(
                startWith(true),
                switchMap(() => this.contactPersonService.getContactPerson(schoolId))
            )
        ]).subscribe(([schoolData, directorData, rentalCostData, contactPerson]) => {

            schoolDetailsData = { ...schoolData, director: directorData, rentalCost: rentalCostData, contactPerson };

            if (this.dialogRef){
                this.dialogRef.componentInstance.data = schoolDetailsData;
            }
            else {
                this.dialogRef = this.dialog.open(SchoolDetailsComponent, {
                    data: schoolDetailsData,
                    disableClose: true
                });
            }

            this.dialogRef.afterClosed().subscribe(() => {
                this.dialogRef = null;
            })

        });
    }

}

export class InterviewFiszki implements IShowDetails {

    private interviewsService: InterviewsService;

    private dialogRef: MatDialogRef<InterviewFiszkiComponent> | null = null;

    constructor(private injector: Injector) {
        this.interviewsService = this.injector.get(InterviewsService);
        // this.fiszkaService = this.injector.get(FiszkaService);
        this.dialog = this.injector.get(MatDialog);
    }


    showDetails(interviewId: string): void {

        console.log('interview ' + interviewId);


        this.interviewsService.getInterview(interviewId).subscribe((interviewData: IInterviewsModel) => {

            if (this.dialogRef)
                this.dialogRef.componentInstance.interview = interviewData;
            else {
                this.dialogRef = this.dialog.open(InterviewFiszkiComponent, {
                    data: interviewData,
                    disableClose: true
                });
            }

            this.dialogRef.afterClosed().subscribe(() => {
                this.dialogRef = null;
            })

        })

    }

}

export class GroupDetails implements IShowDetails {

    private groupService: GroupsService;
    private dialog: MatDialog;

    constructor(private injector: Injector) {
        this.groupService = this.injector.get(GroupsService);
        this.dialog = this.injector.get(MatDialog);
    }

    showDetails(groupId: string): void {

        console.log('group ' + groupId);

        this.groupService.getGroupById(groupId).subscribe({
            next: (groupDetailsData: IGroupModel) => {
                this.dialog.open(GroupFiszkiComponent, {
                    data: groupDetailsData,
                    disableClose: true
                });
            }
        });

    }

}
