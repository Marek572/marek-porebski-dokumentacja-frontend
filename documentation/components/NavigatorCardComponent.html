import { Component, EventEmitter, Input, Output } from '@angular/core';

@Component({
  selector: 'app-navigator-card',
  templateUrl: './navigator-card.component.html',
  styleUrls: ['./navigator-card.component.scss']
})
export class NavigatorCardComponent {

  @Input() tabsLabels: string[] = [];
  selected = 0;
  @Output() tabChange = new EventEmitter<string>();


  onTabChanged(index: number) {
    this.selected = index;
    this.tabChange.emit(this.tabsLabels[index]);
  }

}

<mat-tab-group  [selectedIndex]="selected" (selectedIndexChange)="onTabChanged($event)">
  @for (label of tabsLabels; track $index) {
    <mat-tab class="mat-tab-custom" [label]="label"></mat-tab>
  }
</mat-tab-group>
