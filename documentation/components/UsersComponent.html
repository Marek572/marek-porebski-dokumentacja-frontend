import { Component, Injector, OnInit, ViewChild } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { MatPaginator } from '@angular/material/paginator';
import { MatSort } from '@angular/material/sort';
import { MatTableDataSource } from '@angular/material/table';
import { BehaviorSubject, combineLatest, map, startWith, switchMap } from 'rxjs';
import { EditUserFormComponent } from 'src/app/common/forms/edit-user-form/edit-user-form.component';
import { UserFormComponent } from 'src/app/common/forms/user-form/user-form.component';
import { IUpdateUserModel } from 'src/app/models/user/update-user.model';
import { IUserModel } from 'src/app/models/user/user.model';
import { UsersService } from 'src/app/services/users/users.service';
import { UserDetails } from 'src/app/utils/show-details';

@Component({
  selector: 'app-users',
  templateUrl: './users.component.html',
  styleUrls: ['./users.component.scss']
})
export class UsersComponent implements OnInit {

  constructor(
    private userService: UsersService,
    public dialog: MatDialog,
    public injector: Injector
  ) { }
  isMenuOpen: boolean = false;

  onMenuToggle(): void {
    this.isMenuOpen = !this.isMenuOpen;
  }

  onMenuMouseLeave(): void {
    if (this.isMenuOpen) {
      this.isMenuOpen = false;
    }
  }
  dataSource: MatTableDataSource<IUserModel> = new MatTableDataSource();
  userDetails = new UserDetails(this.injector);
  displayedColumns: string[] = ['name', 'surname', 'phone', 'email', 'role', 'job', 'actions'];

  @ViewChild(MatPaginator) paginator!: MatPaginator;
  @ViewChild('empTbSort') empTbSort = new MatSort();

  tabsLabels = ['Wszyscy', 'Aktywni', 'Nieaktywni'];
  activeTab = new BehaviorSubject<string>('Wszyscy');

  ngOnInit() {
    combineLatest([
      this.userService.usersHasChanged.pipe(
        startWith(true),
        switchMap(() => this.userService.getUsers())
      ),
      this.activeTab
    ]
    ).pipe(
      map(([users, activeTab]) => this.filterUsersByTab(users, activeTab))
    ).subscribe(filteredUsers => {
      this.dataSource.data = filteredUsers;
    });
  }

  ngAfterViewInit() {
    this.dataSource.sort = this.empTbSort;
    this.dataSource.paginator = this.paginator;
  }

  filterUsersByTab(users: IUserModel[], tab: string): IUserModel[] {
    switch (tab) {
      case 'Aktywni':
        return users.filter(user => user.active);
      case 'Nieaktywni':
        return users.filter(user => !user.active);
      case 'Wszyscy':
      default:
        return users;
    }
  }

  onTabChange(tabLabel: string) {
    this.activeTab.next(tabLabel);
  }

  searchbarFilter(inputValue: string) {
    this.dataSource.filter = inputValue.trim().toLowerCase();
    this.dataSource.data = [...this.dataSource.data];
  }

  addBtnClick() {
    this.dialog.open(UserFormComponent, { disableClose: true });
  }

  editBtnClick(event: Event, user: IUpdateUserModel): void {
    event.stopPropagation();
    this.dialog.open(EditUserFormComponent, {
      data: user,
      disableClose: true
    });
  }

  showDetails(userId: string): void {
    this.userDetails.showDetails(userId)
  }

}
