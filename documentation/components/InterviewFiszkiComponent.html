import { Component, Inject, OnInit } from '@angular/core';
import { MAT_DIALOG_DATA, MatDialog, MatDialogRef } from '@angular/material/dialog';
import { IUpdateFiszkaModel } from 'src/app/models/fiszka/update-fiszka.model';
import { FiszkaFormComponent } from '../../forms/fiszka-form/fiszka-form.component';
import { IInterviewsModel } from 'src/app/models/interviews/interviews.model';
import { EditFiszkaFormComponent } from '../../forms/edit-fiszka-form/edit-fiszka-form.component';
import { FiszkaService } from 'src/app/services/fiszka/fiszka.service';
import { MatSnackBar } from '@angular/material/snack-bar';
import { MatTableDataSource } from '@angular/material/table';
import { IFiszkaModel } from 'src/app/models/fiszka/fiszka.model';
import { InterviewsService } from 'src/app/services/interviews/interviews.service';
import { BehaviorSubject, Observable, of, startWith, switchMap } from 'rxjs';
import { FiszkaDetailsComponent } from '../fiszka-details/fiszka-details.component';
import { State } from 'src/app/models/interviews/enum.model';

@Component({
  selector: 'app-interview-fiszki',
  templateUrl: './interview-fiszki.component.html',
  styleUrls: ['./interview-fiszki.component.scss']
})
export class InterviewFiszkiComponent implements OnInit {

  constructor(
    @Inject(MAT_DIALOG_DATA) public interview: IInterviewsModel,
    public dialog: MatDialog,
    public dialogRef: MatDialogRef<InterviewFiszkiComponent>,
    public fiszkaService: FiszkaService,
    public interviewService: InterviewsService,
    private snackBar: MatSnackBar,
  ) { }

  dataSource: MatTableDataSource<IFiszkaModel> = new MatTableDataSource<IFiszkaModel>();

  displayedColumns: string[] = ['name', 'surname', 'class', 'phone', 'email', 'assignedBy', 'actions'];

  ngOnInit() {

    console.log('details data', this.interview);

    //TYMCZASOWE ROZWIAZANIE, NIE WIEM JAK LAPAC TO W SHOW-DETAILS BO SIE NIE UPDATEUJE
    this.fiszkaService.fiszkiHasChanged.pipe(
      startWith(true),
      switchMap(() => this.fiszkaService.getFiszkiByInterviewId(this.interview.id)),
    ).subscribe((fiszki) => {
      this.dataSource.data = fiszki;
    });

  }

  closeBtnClick(): void {
    this.dialogRef.close();
  }

  searchbarFilterFiszki(inputValue: any) {
    this.dataSource.filter = inputValue;
    this.dataSource.data = [...this.dataSource.data];
  }

  addBtnClick(): void {
    console.log('add fiszka');
    this.dialog.open(FiszkaFormComponent, {
      data: this.interview.id,
      disableClose: true
    });
  }

  submitBtnClick(): void {
    console.log('submit wywiadowka');
    this.interviewService.updateInterview({ state: State.APROVE }, this.interview.id).subscribe({
      next: () => { },
      complete: () => {
        this.snackBar.open('Wywiadówka zatwierdzona', '', {
          duration: 4000,
          panelClass: ['custom-snackbar']
        });
        this.dialogRef.close();
      }
    });
  }

  rejectBtnClick() : void {
    console.log('reject wywiadowka');
    this.interviewService.updateInterview({ state: State.REJECT }, this.interview.id).subscribe({
      next: () => { },
      complete: () => {
        this.snackBar.open('Wywiadówka odrzucona', '', {
          duration: 4000,
          panelClass: ['custom-snackbar']
        });
        this.dialogRef.close();
      }
    });
  }


  editBtnClick(event: Event, fiszka: IUpdateFiszkaModel) {
    console.log('edit fiszka', fiszka);
    event.stopPropagation();
    this.dialog.open(EditFiszkaFormComponent, {
      data: fiszka,
      disableClose: true
    });
  }

  deleteBtnClick(event: Event,fiszkaId: string) {
    console.log('delete fiszka', fiszkaId);
    event.stopPropagation();
    this.fiszkaService.deleteFiszka(fiszkaId).subscribe({
      next: () => { },
      error: (err) => { console.log(err); },
      complete: () => {
        this.snackBar.open('Fiszka usunięta', '', {
          duration: 4000,
          panelClass: ['custom-snackbar']
        });
      }
    });
  }

  fiszkaDetails(fiszkaData: IFiszkaModel) {
    this.dialog.open(FiszkaDetailsComponent, {
      data: fiszkaData,
      disableClose: true
    });
  }

}
