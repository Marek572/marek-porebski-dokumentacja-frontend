import { Component, Injector, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { MatTableDataSource } from '@angular/material/table';
import { startWith, switchMap } from 'rxjs';
import { FiszkaDetailsComponent } from 'src/app/common/details/fiszka-details/fiszka-details.component';
import { EditFiszkaFormComponent } from 'src/app/common/forms/edit-fiszka-form/edit-fiszka-form.component';
import { IFiszkaModel } from 'src/app/models/fiszka/fiszka.model';
import { IUpdateFiszkaModel } from 'src/app/models/fiszka/update-fiszka.model';
import { ISchoolModel } from 'src/app/models/school/school.model';
import { FiszkaService } from 'src/app/services/fiszka/fiszka.service';
import { SchoolsService } from 'src/app/services/schools/schools.service';

@Component({
  selector: 'app-fiszki',
  templateUrl: './fiszki.component.html',
  styleUrls: ['./fiszki.component.scss']
})
export class FiszkiComponent implements OnInit {

  constructor(
    public dialog: MatDialog,
    private schoolService: SchoolsService,
    public fiszkaService: FiszkaService,
    public injector: Injector
  ) { }

  schoolList: ISchoolModel[] = [];

  displayedColumns: string[] = ['name', 'surname', 'class', 'phone', 'email', 'assignedBy', 'actions'];
  dataSource: MatTableDataSource<IFiszkaModel> = new MatTableDataSource<IFiszkaModel>();

  ngOnInit() {

    this.schoolService.getSchools().subscribe({
      next: (schools: ISchoolModel[]) => {
        this.schoolList = schools;
      }
    })

    this.fiszkaService.fiszkiHasChanged.pipe(
      startWith(true),
      switchMap(() => this.fiszkaService.getFiszkiByRegion()),
    ).subscribe((fiszki: IFiszkaModel[]) => {
      this.dataSource.data = fiszki;
      console.log(fiszki);
    });

  }

  test(event: Event): void {
    console.log(event);
  }

  searchbarFilterFiszki(inputValue: string) {
    this.dataSource.filter = inputValue.trim().toLowerCase();
    this.dataSource.data = [...this.dataSource.data];
  }

  fiszkaDetails(fiszkaData: IFiszkaModel) {
    this.dialog.open(FiszkaDetailsComponent, {
      data: fiszkaData,
      disableClose: true
    });
  }

  editBtnClick(event: Event, fiszka: IUpdateFiszkaModel) {
    event.stopPropagation();
    this.dialog.open(EditFiszkaFormComponent, {
      data: fiszka,
      disableClose: true
    });
  }

}
