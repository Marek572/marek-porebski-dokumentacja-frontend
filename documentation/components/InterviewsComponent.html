import { Component, Injector, OnInit, ViewChild } from '@angular/core';
import { InterviewsService } from 'src/app/services/interviews/interviews.service';
import { MatDialog } from '@angular/material/dialog';
import { BehaviorSubject, combineLatest, map, startWith, switchMap } from 'rxjs';
import { IInterviewsModel } from 'src/app/models/interviews/interviews.model';
import { IUpdateInterviewsModel } from 'src/app/models/interviews/update-interviews.model';
import { InterviewFormComponent } from 'src/app/common/forms/interview-form/interview-form.component';
import { EditInterviewFormComponent } from 'src/app/common/forms/edit-interview-form/edit-interview-form.component';
import { InterviewFiszki } from 'src/app/utils/show-details';
import { MatTableDataSource } from '@angular/material/table';
import { MatPaginator } from '@angular/material/paginator';
import { MatSort } from '@angular/material/sort';
import { State } from 'src/app/models/interviews/enum.model';

@Component({
  selector: 'app-interviews',
  templateUrl: './interviews.component.html',
  styleUrl: './interviews.component.scss'
})
export class InterviewsComponent implements OnInit {

  constructor(
    private interviewsService: InterviewsService,
    public dialog: MatDialog,
    public injector: Injector
  ) { }

  dataSource: MatTableDataSource<IInterviewsModel> = new MatTableDataSource();
  interviewFiszki = new InterviewFiszki(this.injector)
  displayedColumns: string[] = ['school', 'date', 'classNumber', 'termsNumber', 'interviewTotalFiszki', 'actions'];

  @ViewChild(MatPaginator) paginator!: MatPaginator;
  @ViewChild('empTbSort') empTbSort = new MatSort();

  tabsLabels = ['Wszystkie', 'Do przypisania', 'Otwarte', 'Do sprawdzenia', 'Odrzucone', 'Zakończone']
  activeTab = new BehaviorSubject<string>('Wszystkie');

  ngOnInit(): void {
    combineLatest([
      this.interviewsService.interviewsHasChanged.pipe(
        startWith(true),
        switchMap(() => this.interviewsService.getInterviews()),
      ),
      this.activeTab
    ]
    ).pipe(
      map(([interviews, activeTab]) => this.filterInterviewsByTab(interviews, activeTab))
    ).subscribe(filteredInterviews => {
      this.dataSource.data = filteredInterviews;
    });
  }

  ngAfterViewInit(): void {
    this.dataSource.sort = this.empTbSort;
    this.dataSource.paginator = this.paginator;
  }

  filterInterviewsByTab(interviews: IInterviewsModel[], tab: string): IInterviewsModel[] {
    switch (tab) {
      case 'Zakończone':
        return interviews.filter(interviews => interviews.confirmed === true);
      case 'Odrzucone':
        return interviews.filter(interviews => interviews.state === State.REJECT);
      case 'Do sprawdzenia':
        return interviews.filter(interviews => interviews.state === State.WAITINGFORAPROVE);
      case 'Otwarte':
        return interviews.filter(interviews => interviews.state === State.WIP);
      case 'Do przypisania':
        return interviews.filter(interviews => interviews.userInterviews.length === 0);
      // case 'Wszystkie':
      default:
        return interviews;
    }
  }

  onTabChange(tabLabel: string): void {
    this.activeTab.next(tabLabel);
  }

  searchbarFilter(inputValue: string) {
    this.dataSource.filter = inputValue.trim().toLowerCase();
    this.dataSource.data = [...this.dataSource.data];
  }

  addBtnClick(): void {
    console.log('add new');
    this.dialog.open(InterviewFormComponent, { disableClose: true });
  }

  editBtnClick(event: Event, formData: IUpdateInterviewsModel): void {
    event.stopPropagation();
    this.dialog.open(EditInterviewFormComponent, {
      data: formData,
      disableClose: true
    });
  }

  showDetails(interviewId: string): void {
    this.interviewFiszki.showDetails(interviewId)
  }

}
