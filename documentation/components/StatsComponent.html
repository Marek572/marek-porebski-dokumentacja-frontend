import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { ChartDataset, ChartOptions } from 'chart.js';
import { IErrorAPIModel } from 'src/app/models/errorAPI.model';
import { ISchoolModel } from 'src/app/models/school/school.model';
import { ISchoolInRegionStatsModel, IStatsModel } from 'src/app/models/stats/stats.model';
import { IUserModel } from 'src/app/models/user/user.model';
import { SchoolsService } from 'src/app/services/schools/schools.service';
import { UsersService } from 'src/app/services/users/users.service';

@Component({
  selector: 'app-stats',
  templateUrl: './stats.component.html',
  styleUrls: ['./stats.component.scss']
})
export class StatsComponent implements OnInit {

  constructor(
    public userService: UsersService,
    public schoolService: SchoolsService,
    private formBuilder: FormBuilder,
  ) { }

  userStatsForm!: FormGroup;
  schoolStatsForm!: FormGroup;
  regionStatsForm!: FormGroup;

  usersList: IUserModel[] = [];
  schoolsList: ISchoolModel[] = [];

  userStats: IStatsModel[] = [];
  userChartLabels: string[] = []
  userChartData: ChartDataset[] = [];

  schoolStats: IStatsModel[] = [];
  schoolChartLabels: string[] = []
  schoolChartData: ChartDataset[] = [];

  regionStats: ISchoolInRegionStatsModel[] = [];
  regionChartLabels: string[] = []
  regionChartData: ChartDataset[] = [];

  chartOptions: ChartOptions = {
    color: 'black',
    responsive: true,
    scales: {
      x: {
        title: {
          display: true,
          text: 'Data wywiadówki',
          color: 'black',
          font: {
            size: 15,
            weight: 'bold'
          }
        },
        ticks: {
          maxRotation: 0,
          minRotation: 0,
          color: 'black'
        },
        grid: {
          color: 'black'
        }
      },
      y: {
        title: {
          display: true,
          text: 'Zebrane fiszki',
          color: 'black',
          font: {
            size: 15,
            weight: 'bold'
          }
        },
        beginAtZero: true,
        ticks: {
          precision: 0,
          color: 'black'
        },
        grid: {
          color: 'black'
        }
      },
    },
    plugins: {
      colors: {
        enabled: true
      },
    }
  };

  regionChartOptions = JSON.parse(JSON.stringify(this.chartOptions));
  

  ngOnInit() {

    this.regionChartOptions.scales.x.title.text = 'Nazwy placówek';

    this.userStatsForm = this.formBuilder.group({
      user: ['', Validators.required],
      startDate: ['', Validators.required],
      endDate: ['', Validators.required],
    }, { updateOn: 'blur' })

    this.userService.getUsers().subscribe((users: IUserModel[]) => {
      this.usersList = users;
    })

    this.schoolStatsForm = this.formBuilder.group({
      school: ['', Validators.required],
      startDate: ['', Validators.required],
      endDate: ['', Validators.required],
    }, { updateOn: 'blur' })

    this.schoolService.getSchools().subscribe((schools: ISchoolModel[]) => {
      this.schoolsList = schools;
    })

    this.regionStatsForm = this.formBuilder.group({
      startDate: ['', Validators.required],
      endDate: ['', Validators.required],
    }, { updateOn: 'blur' })

    let regionStartDate: Date = new Date();
    let regionEndDate: Date  = new Date();
    regionStartDate.setMonth(regionEndDate.getMonth() - 11);

    this.userService.getRegionStatsInRange(regionStartDate, regionEndDate)
      .subscribe({
        next: (stats: ISchoolInRegionStatsModel[]) => { this.regionStats = stats },
        error: (err: IErrorAPIModel) => {
          console.log(err);
        },
        complete: () => {
          this.regionChartLabels = this.regionStats.map((stat: ISchoolInRegionStatsModel) => stat.name);
          this.regionChartData = [
            {
              data: this.regionStats.map((stat: ISchoolInRegionStatsModel) => stat.totalFiszki),
              label: 'Zebrane fiszki',
              backgroundColor: 'orange'
            }
          ];
        }
      });

  }

  onUserStatsFormSubmit(): void {

    if (this.userStatsForm.invalid)
      return;

    else {

      const userId = this.userStatsForm.get('user')?.value
      const startDate = this.userStatsForm.get('startDate')?.value
      const endDate = this.userStatsForm.get('endDate')?.value

      this.userService.getUserStatsInRange(startDate, endDate, userId)
        .subscribe({
          next: (stats: IStatsModel[]) => { this.userStats = stats },
          error: (err: IErrorAPIModel) => {
            console.log(err);
          },
          complete: () => {
            this.userChartLabels = this.userStats.map((stat: IStatsModel) => new Date(stat.date).toLocaleDateString('default', { day: '2-digit', month: '2-digit' }));
            this.userChartData = [
              {
                data: this.userStats.map((stat: IStatsModel) => stat.count),
                label: 'Zebrane fiszki',
                backgroundColor: 'orange'
              }
            ];
          }
        })
    }

  }

  onSchoolStatsFormSubmit(): void {

    if (this.schoolStatsForm.invalid)
      return;

    else {

      const schoolId = this.schoolStatsForm.get('school')?.value
      const startDate = this.schoolStatsForm.get('startDate')?.value
      const endDate = this.schoolStatsForm.get('endDate')?.value

      this.userService.getSchoolStatsInRange(schoolId, startDate, endDate)
        .subscribe({
          next: (stats: IStatsModel[]) => { this.schoolStats = stats },
          error: (err: IErrorAPIModel) => {
            console.log(err);
          },
          complete: () => {
            this.schoolChartLabels = this.schoolStats.map((stat: IStatsModel) => new Date(stat.date).toLocaleDateString('default', { day: '2-digit', month: '2-digit' }));
            this.schoolChartData = [
              {
                data: this.schoolStats.map((stat: IStatsModel) => stat.count),
                label: 'Zebrane fiszki',
                backgroundColor: 'orange'
              }
            ];
          }
        })
    }

  }

  onRegionStatsFormSubmit(): void {

    if (this.regionStatsForm.invalid)
      return;

    else {

      const startDate = this.regionStatsForm.get('startDate')?.value
      const endDate = this.regionStatsForm.get('endDate')?.value

      this.userService.getRegionStatsInRange(startDate, endDate)
        .subscribe({
          next: (stats: ISchoolInRegionStatsModel[]) => { this.regionStats = stats, console.log(stats); },
          error: (err: IErrorAPIModel) => {
            console.log(err);
          },
          complete: () => {
            this.regionChartLabels = this.regionStats.map((stat: ISchoolInRegionStatsModel) => stat.name);
            this.regionChartData = [
              {
                data: this.regionStats.map((stat: ISchoolInRegionStatsModel) => stat.totalFiszki),
                label: 'Zebrane fiszki',
                backgroundColor: 'orange'
              }
            ];
          }
        });
    }

  }

}
