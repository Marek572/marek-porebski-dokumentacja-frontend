import { Component, Inject, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialogRef } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { IErrorAPIModel } from 'src/app/models/errorAPI.model';
import { FiszkaService } from 'src/app/services/fiszka/fiszka.service';

@Component({
  selector: 'app-fiszka-form',
  templateUrl: './fiszka-form.component.html',
  styleUrls: ['./fiszka-form.component.scss']
})
export class FiszkaFormComponent implements OnInit {

  constructor(
    @Inject(MAT_DIALOG_DATA) public data: string,
    public dialogRef: MatDialogRef<FiszkaFormComponent>,
    private formBuilder: FormBuilder,
    public fiszkaService: FiszkaService,
    private snackBar: MatSnackBar
  ) { }

  fiszkaForm!: FormGroup;

  ngOnInit() {

    this.fiszkaForm = this.formBuilder.group({
      name: ['', Validators.required],
      surname: ['', Validators.required],
      classDigit: ['', Validators.compose([
        Validators.maxLength(1),
        Validators.min(0),
        Validators.max(8)
      ])],
      classLetter: ['', Validators.compose([
        Validators.maxLength(1),
        Validators.pattern('^[A-Za-z]$'),
      ])],
      phone: ['', Validators.compose([
        Validators.required,
        Validators.pattern(new RegExp('^(?=.*[0-9]).{9,9}$'))
      ])],
      email: ['', Validators.email],
      comments: ['',]
    }, { updateOn: 'blur' });

  }

  onFormSubmit() {
    console.log(this.fiszkaForm.value);
    console.log('interview id', this.data);

    if (this.fiszkaForm.invalid)
      return;

    else {
      this.fiszkaService.addFiszkaToInterview(this.fiszkaForm.value, this.data)
        .subscribe({
          next: () => { },
          error: (err: IErrorAPIModel) => {
            console.log(err);
          },
          complete: () => {
            this.snackBar.open('Fiszka dodana pomyślnie', '', {
              duration: 4000,
              panelClass: ['custom-snackbar']
            });
            this.dialogRef.close();
          }
        })
    }

  }

  onFormCancel() {
    this.dialogRef.close();
  }

}
