import { Component, Inject, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialog, MatDialogRef } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { IPhoneCallsModel } from 'src/app/models/phone-calls/phone-calls.model';
import { FiszkaService } from 'src/app/services/fiszka/fiszka.service';
import { PhoneCallService } from 'src/app/services/phone-call/phone-call.service';

@Component({
  selector: 'app-phone-call-form',
  templateUrl: './phone-call-form.component.html',
  styleUrls: ['./phone-call-form.component.scss'],
})
export class PhoneCallFormComponent implements OnInit {

  constructor(
    @Inject(MAT_DIALOG_DATA) public fiszkaId: string,
    public phoneCallService: PhoneCallService,
    private fiszkaService: FiszkaService,
    public dialogRef: MatDialogRef<PhoneCallFormComponent>,
    public dialog: MatDialog,
    private formBuilder: FormBuilder,
    private snackBar: MatSnackBar,
  ) { }

  phoneCallForm!: FormGroup;
  currentDate: Date = new Date();

  ngOnInit(): void {

    this.phoneCallForm = this.formBuilder.group({
      callDate: [this.currentDate, Validators.required],
      callNext: [''],
      comments: ['', Validators.required],
    })

  }

  onFormCancel(): void {

    this.dialogRef.close();

  }

  onFormSubmit(): void {

    console.log(this.phoneCallForm.value);

    const callDate: Date = new Date(this.phoneCallForm.get('callDate')?.value);
    const callNext: Date = new Date(this.phoneCallForm.get('callNext')?.value);
    const comments = this.phoneCallForm.get('comments')?.value;

    if (this.phoneCallForm.invalid)
      return
    else {
      this.phoneCallService.addPhoneCall(this.fiszkaId, {callDate, callNext, comments}).subscribe({
        next: () => {},
        error: (err) => {console.log(err);},
        complete: () => {
          this.snackBar.open('Rozmowa telefoniczna dodana pomyślnie', '', {
            duration: 4000,
            panelClass: ['custom-snackbar']
          });
          this.fiszkaService.fiszkiHasChanged.next(true);
          this.dialogRef.close();
        }
      })
    }

  }

}
