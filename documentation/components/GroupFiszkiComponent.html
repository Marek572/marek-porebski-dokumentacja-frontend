import { Component, Inject, OnChanges, OnInit, SimpleChanges } from '@angular/core';
import { MAT_DIALOG_DATA, MatDialog } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { MatTableDataSource } from '@angular/material/table';
import { IFiszkaModel } from 'src/app/models/fiszka/fiszka.model';
import { IGroupModel } from 'src/app/models/group/group.model';
import { FiszkaService } from 'src/app/services/fiszka/fiszka.service';
import { GroupsService } from 'src/app/services/groups/groups.service';
import { GroupFiszkaFormComponent } from '../../forms/group-fiszka-form/group-fiszka-form.component';
import { FiszkaDetailsComponent } from '../fiszka-details/fiszka-details.component';
import { startWith, switchMap } from 'rxjs';

@Component({
  selector: 'app-group-fiszki',
  templateUrl: './group-fiszki.component.html',
  styleUrls: ['./group-fiszki.component.scss']
})
export class GroupFiszkiComponent implements OnInit  {

  constructor(
    @Inject(MAT_DIALOG_DATA) public group: IGroupModel,
    public dialog: MatDialog,
    public groupService: GroupsService,
    public fiszkaService: FiszkaService,
    private snackBar: MatSnackBar,
  ) { }

  displayedColumns: string[] = ['name', 'surname', 'class', 'phone', 'email', 'assignedBy', 'actions'];
  dataSource: MatTableDataSource<IFiszkaModel> = new MatTableDataSource<IFiszkaModel>();

  ngOnInit(): void {

    //TYMCZASOWE ROZWIAZANIE, NIE WIEM JAK LAPAC TO W SHOW-DETAILS BO SIE NIE UPDATEUJE
    this.fiszkaService.fiszkiHasChanged.pipe(
      startWith(true),
      switchMap(() => this.fiszkaService.getFiszkiByGroupId(this.group.id)),
    ).subscribe((fiszki) => {
      this.dataSource.data = fiszki;
    });

  }

  closeBtnClick(): void {
    this.dialog.closeAll();
  }

  searchbarFilterFiszki(inputValue: string) {
    this.dataSource.filter = inputValue.trim().toLowerCase();
    this.dataSource.data = [...this.dataSource.data];
  }

  addBtnClick(): void {
    console.log('add fiszka');
    this.dialog.open(GroupFiszkaFormComponent, {
      data: this.group,
      disableClose: true
    });
  }

  deleteBtnClick(event: Event, fiszkaId: string) {
    event.stopPropagation();
    this.fiszkaService.deleteFiszkaFromGroup(fiszkaId, this.group.id).subscribe({
      next: () => {},
      error: (err) => {console.log(err);},
      complete: () => {
        this.snackBar.open('Fiszka usunięta', '', {
          duration: 4000,
          panelClass: ['custom-snackbar']
        });
      }
    });
  }

  fiszkaDetails(fiszkaData: IFiszkaModel) {
    this.dialog.open(FiszkaDetailsComponent, {
      data: fiszkaData,
      disableClose: true
    });
  }

}
