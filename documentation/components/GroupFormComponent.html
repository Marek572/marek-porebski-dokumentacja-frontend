import { Component, Inject, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialog } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { DaysOfTheWeek } from 'src/app/models/daysOfTheWeek.model';
import { IErrorAPIModel } from 'src/app/models/errorAPI.model';
import { Subject } from 'src/app/models/group/enum.model';
import { GroupsService } from 'src/app/services/groups/groups.service';
import { DataLossDialogComponent } from '../../dialog/dataLoss-dialog/dataLoss-dialog/dataLoss-dialog.component';

@Component({
  selector: 'app-group-form-component',
  templateUrl: './group-form.component.html',
  styleUrls: ['./group-form.component.scss']
})
export class GroupFormComponent implements OnInit {

  constructor(
    public dialog: MatDialog,
    @Inject(MAT_DIALOG_DATA) public schoolId: string,
    private formBuilder: FormBuilder,
    private groupService: GroupsService,
    private snackBar: MatSnackBar,
  ) { }

  groupFrom!: FormGroup;
  daysOfTheWeekList = DaysOfTheWeek;
  subjectList = Subject;

  ngOnInit() {

    this.groupFrom = this.formBuilder.group({
      title: ['', Validators.required],
      codeGroup: ['', Validators.required],
      subject: ['', Validators.required],
      dayOfTheWeek: [''],
      hour: [''],
    })

  }

  doNotSort() {return 0}

  onFormSubmit() {
    if (this.groupFrom.invalid)
      return;

    else {
      this.groupService.addGroup(this.groupFrom.value, this.schoolId)
        .subscribe({
          next: () => {},
          error: (err: IErrorAPIModel)=>{
            console.log(err);
            // handleError(this.userForm,err);
          },
          complete: ()=>{
            this.snackBar.open('Grupa dodana pomyślnie', '', {
              duration: 4000,
              panelClass: ['custom-snackbar']
            });
            this.dialog.closeAll();
          }
        })
    }
  }

  onFormCancel() {
   
    if (!this.groupFrom.dirty) {
      this.dialog.closeAll();
    }else{
      let dialogRef = this.dialog.open(DataLossDialogComponent, {});
    }
  }

}
