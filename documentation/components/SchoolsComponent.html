import { Component, Injector, OnInit, ViewChild } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { BehaviorSubject, combineLatest, map, startWith, switchMap } from 'rxjs';
import { SchoolFormComponent } from 'src/app/common/forms/school-form/school-form.component';
import { EditSchoolFormComponent } from 'src/app/common/forms/edit-school-form/edit-school-form.component';
import { ISchoolModel } from 'src/app/models/school/school.model';
import { SchoolsService } from 'src/app/services/schools/schools.service';
import { SchoolDetails } from 'src/app/utils/show-details';
import { MatTableDataSource } from '@angular/material/table';
import { MatPaginator } from '@angular/material/paginator';
import { MatSort } from '@angular/material/sort';

@Component({
  selector: 'app-schools',
  templateUrl: './schools.component.html',
  styleUrls: ['./schools.component.scss']
})
export class SchoolsComponent implements OnInit {

  constructor(
    private schoolService: SchoolsService,
    public dialog: MatDialog,
    public injector: Injector
  ) { }

  dataSource: MatTableDataSource<ISchoolModel> = new MatTableDataSource();
  schoolDetails = new SchoolDetails(this.injector)
  displayedColumns: string[] = ['name', 'status', 'type', 'street', 'actions'];

  @ViewChild(MatPaginator) paginator!: MatPaginator;
  @ViewChild('empTbSort') empTbSort = new MatSort();

  tabsLabels = ['Wszystkie', 'Aktywne', 'Nieaktywne'];
  activeTab = new BehaviorSubject<string>('Wszyscy');

  ngOnInit(): void {
    combineLatest([
      this.schoolService.schoolsHasChanged.pipe(
        startWith(true),
        switchMap(() => this.schoolService.getSchools()),
      ),
      this.activeTab
    ]
    ).pipe(
      map(([schools, activeTab]) => this.filterSchoolsByTab(schools, activeTab))
    ).subscribe(filteredSchools => {
      this.dataSource.data = filteredSchools;
    });
  }

  ngAfterViewInit(): void {
    this.dataSource.sort = this.empTbSort;
    this.dataSource.paginator = this.paginator;
  }

  filterSchoolsByTab(schools: ISchoolModel[], tab: string): ISchoolModel[] {
    switch (tab) {
      case 'Nieaktywne':
        return schools.filter(schools => schools.cooperation === false);
      case 'Aktywne':
        return schools.filter(schools => schools.cooperation === true);
      // case 'Wszystkie':
      default:
        return schools;
    }
  }

  onTabChange(tabLabel: string): void {
    this.activeTab.next(tabLabel);
  }

  searchbarFilter(inputValue: string) {
    this.dataSource.filter = inputValue.trim().toLowerCase();
    this.dataSource.data = [...this.dataSource.data];
  }

  addBtnClick(): void {
    console.log('add new');
    this.dialog.open(SchoolFormComponent, { disableClose: true });
  }

  editBtnClick(event: Event, formData: ISchoolModel): void {
    event.stopPropagation();
    this.dialog.open(EditSchoolFormComponent, {
      data: formData,
      disableClose: true
    });
  }

  showDetails(schoolId: string): void {
    this.schoolDetails.showDetails(schoolId)
  }

}
