import { Component, Inject } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { Router } from '@angular/router';
import { Observable } from 'rxjs';
import { ConfirmDialogComponent } from 'src/app/common/dialog/confirm-dialog/confirm-dialog.component';
import { IProfileModel } from 'src/app/models/profile/profile.model';
import { IUserModel } from 'src/app/models/user/user.model';
import { IErrorAPIModel } from 'src/app/models/errorAPI.model';
import { AuthService } from 'src/app/services/auth/auth.service';
import { UsersService } from 'src/app/services/users/users.service';
import { handleError } from 'src/app/utils/common-functions';
import { IStatsModel } from 'src/app/models/stats/stats.model';
import { ChartDataset, ChartOptions } from 'chart.js';

@Component({
  selector: 'app-profile',
  templateUrl: './profile.component.html',
  styleUrls: ['./profile.component.scss']
})
export class ProfileComponent {

  constructor(
    private formBuilder: FormBuilder,
    @Inject(Router) private router: Router,
    private auth: AuthService,
    public userService: UsersService,
    private dialog: MatDialog
  ) { }

  loggedUser = new Observable<IUserModel>;
  phoneForm!: FormGroup;
  emailForm!: FormGroup;
  loggedUserData: IProfileModel = {
    id: '',
    name: '',
    surname: '',
    phone: '',
    email: '',
    region: '',
    role: '',
    job: ''
  };

  loggedUserStats: IStatsModel[] = [];
  chartLabels: string[] = []
  chartData: ChartDataset[] = [];

  chartOptions: ChartOptions = {
    color: 'black',
    responsive: true,
    scales: {
      x: {
        title: {
          display: true,
          text: 'Data wywiadówki',
          color: 'black',
          font: {
            size: 15,
            weight: 'bold'
          }
        },
        ticks: {
          maxRotation: 0,
          minRotation: 0,
          color: 'black'
        },
        grid: {
          color: 'black'
        }
      },
      y: {
        title: {
          display: true,
          text: 'Fiszki',
          color: 'black',
          font: {
            size: 15,
            weight: 'bold'
          }
        },
        beginAtZero: true,
        ticks: {
          precision: 0,
          color: 'black'
        },
        grid: {
          color: 'black'
        }
      },
    },
    plugins: {
      colors: {
        enabled: false
      },
    }
  };

  ngOnInit(): void {

    this.phoneForm = this.formBuilder.group({
      phoneInput: ['', Validators.compose([
        Validators.required,
        Validators.pattern(new RegExp('^(?=.*[0-9]).{9,9}$'))])
      ],
    })

    this.emailForm = this.formBuilder.group({
      emailInput: ['', Validators.compose([
        Validators.required,
        Validators.email])
      ],
    })

    this.loggedUser = this.auth.getProfile()
    this.loggedUser.subscribe({
      next: (data: any) => { this.loggedUserData = data },
      complete: () => {
        this.phoneForm.get('phoneInput')?.setValue(this.loggedUserData.phone);
        this.emailForm.get('emailInput')?.setValue(this.loggedUserData.email);
      }
    })

    //FIXME: daty testowe, do zmiany na koniec
    // let startDate = new Date();
    let startDate = new Date('2023-05-01T06:35:22.000Z');
    let endDate = new Date();
    // endDate.setMonth(startDate.getMonth() - 1);

    this.userService.getUserStatsInRange(startDate, endDate).subscribe({
      next: (stats: IStatsModel[]) => this.loggedUserStats = stats,
      complete: () => {
        this.chartLabels = this.loggedUserStats.map((stat: IStatsModel) => new Date(stat.date).toLocaleDateString('default', { day: '2-digit', month: '2-digit' }));
        this.chartData = [
          {
            data: this.loggedUserStats.map((stat: IStatsModel) => stat.count),
            label: 'Zebrane fiszki',
            backgroundColor: 'orange'
          }
        ];
      }
    });

  }

  isPhoneReadonly = true;
  phoneReadonlySwitch() {
    this.isPhoneReadonly = !this.isPhoneReadonly
  }
  onPhoneEditClick() {
    this.phoneReadonlySwitch()
  }
  onPhoneCancelClick() {
    this.phoneForm.get('phoneInput')?.setValue(this.loggedUserData.phone);
    this.phoneReadonlySwitch()
  }
  onPhoneDoneClick() {

    const newPhone = this.phoneForm.get('phoneInput')?.value

    if (this.phoneForm.invalid)
      return;

    if (newPhone === this.loggedUserData.phone)
      return this.phoneReadonlySwitch();

    console.log('updateSelfPhone', newPhone, this.loggedUserData.id);
    this.userService.updateSelf({ phone: newPhone }, this.loggedUserData.id)
      .subscribe({
        next: () => { },
        error: (err: IErrorAPIModel) => {
          handleError(this.phoneForm, err, 'phoneInput')
        },
        complete: () => {
          this.phoneReadonlySwitch();
        }
      })

  }

  isEmailReadonly = true;
  emailReadonlySwitch() {
    this.isEmailReadonly = !this.isEmailReadonly
  }
  onEmailEditClick() {
    this.emailReadonlySwitch()
    console.log('edit');
  }
  onEmailCancelClick() {
    this.emailForm.get('emailInput')?.setValue(this.loggedUserData.email);
    this.emailReadonlySwitch()
  }
  onEmailDoneClick() {

    const newEmail = this.emailForm.get('emailInput')?.value

    if (this.emailForm.invalid)
      return;

    if (newEmail === this.loggedUserData.email)
      return this.emailReadonlySwitch();
    const dialogRef = this.dialog.open(ConfirmDialogComponent);
    dialogRef.afterClosed().subscribe(result => {
      if (result) {
        this.userService.updateSelf({ email: newEmail }, this.loggedUserData.id)
          .subscribe({
            next: () => { },
            error: (err: IErrorAPIModel) => {
              handleError(this.emailForm, err, 'emailInput')
            },
            complete: () => {
              this.emailReadonlySwitch()
              this.auth.logout();
              this.router.navigate(['/'])
            }
          })
      }
    });

  }
}
