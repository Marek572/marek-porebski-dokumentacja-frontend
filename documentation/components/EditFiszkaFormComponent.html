import { Component, Inject, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialogRef } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { IErrorAPIModel } from 'src/app/models/errorAPI.model';
import { IUpdateFiszkaModel } from 'src/app/models/fiszka/update-fiszka.model';
import { FiszkaService } from 'src/app/services/fiszka/fiszka.service';

@Component({
  selector: 'app-edit-fiszka-form',
  templateUrl: './edit-fiszka-form.component.html',
  styleUrls: ['./edit-fiszka-form.component.scss']
})
export class EditFiszkaFormComponent implements OnInit {

  constructor(
    @Inject(MAT_DIALOG_DATA) public editFiszkaData: IUpdateFiszkaModel,
    public dialogRef: MatDialogRef<EditFiszkaFormComponent>,
    private formBuilder: FormBuilder,
    public fiszkaService: FiszkaService,
    private snackBar: MatSnackBar
  ) { }

  fiszkaEditForm!: FormGroup;

  ngOnInit() {

    const { id, name, surname, classDigit, classLetter, phone, email, comments } = this.editFiszkaData

    console.log(id);
    

    this.fiszkaEditForm = this.formBuilder.group({
      name: [name, Validators.required],
      surname: [surname, Validators.required],
      classDigit: [classDigit, Validators.compose([
        Validators.maxLength(1),
        Validators.min(0),
        Validators.max(8)
      ])],
      classLetter: [classLetter, Validators.compose([
        Validators.maxLength(1),
        Validators.pattern('^[A-Za-z]$'),
      ])],
      phone: [phone, Validators.compose([
        Validators.required,
        Validators.pattern(new RegExp('^(?=.*[0-9]).{9,9}$'))
      ])],
      email: [email, Validators.email],
      comments: [comments,]
    }, { updateOn: 'blur' });

  }

  getDirtyData() {
    const dirtyData: any = {};

    Object.keys(this.fiszkaEditForm.controls).forEach(key => {

      const control = this.fiszkaEditForm.get(key);

      if (control?.dirty)
        dirtyData[key] = control.value;

    });

    return dirtyData;
  }

  onFormSubmit() {

    console.log('dirty data', this.getDirtyData());
    console.log(this.editFiszkaData);

    if (this.fiszkaEditForm.invalid)
      return;

    if (this.fiszkaEditForm.dirty) {
      this.fiszkaService.updateFiszka(this.getDirtyData(), this.editFiszkaData.id)
        .subscribe({
          next: () => { },
          error: (err: IErrorAPIModel) => {
            console.log(err);
          },
          complete: () => {
            this.snackBar.open('Fiszka zmieniona pomyślnie', '', {
              duration: 4000,
              panelClass: ['custom-snackbar']
            });
            this.dialogRef.close();
          }
        })
    }

  }

  onFormCancel() {
    this.dialogRef.close();
  }

}
