import { Component, Injector, OnInit } from '@angular/core';
import { MatOptionSelectionChange } from '@angular/material/core';
import { MatDialog } from '@angular/material/dialog';
import { MatTableDataSource } from '@angular/material/table';
import { startWith, switchMap } from 'rxjs';
import { EditGroupFormComponent } from 'src/app/common/forms/edit-group-form/edit-group-form.component';
import { GroupFormComponent } from 'src/app/common/forms/group-form/group-form.component';
import { IGroupModel } from 'src/app/models/group/group.model';
import { ISchoolModel } from 'src/app/models/school/school.model';
import { GroupsService } from 'src/app/services/groups/groups.service';
import { SchoolsService } from 'src/app/services/schools/schools.service';
import { GroupDetails } from 'src/app/utils/show-details';

@Component({
  selector: 'app-groups',
  templateUrl: './groups.component.html',
  styleUrls: ['./groups.component.scss']
})
export class GroupsComponent implements OnInit {

  constructor(
    public dialog: MatDialog,
    private schoolService: SchoolsService,
    private groupsService: GroupsService,
    public injector: Injector
  ) { }

  schoolList: ISchoolModel[] = [];

  dataSource: MatTableDataSource<IGroupModel> = new MatTableDataSource<IGroupModel>;
  displayedColumns: string[] = ['codeGroup', 'subject', 'dayOfTheWeek', 'hour', 'trial', 'actions'];

  groupDetails = new GroupDetails(this.injector)

  ngOnInit() {

    this.schoolService.getSchools().subscribe({
      next: (schools: ISchoolModel[]) => {
        this.schoolList = schools;
      }
    })

  }

  onSelectedSchool(schoolId: string) {

    this.groupsService.groupsHasChanged.pipe(
      startWith(true),
      switchMap(() => this.groupsService.getGroupsBySchoolId(schoolId)),
    ).subscribe({
      next: (groups: IGroupModel[]) => {
        this.dataSource.data = groups;
      },
    });

  }

  searchbarFilterGrupy(inputValue: string) {
    this.dataSource.filter = inputValue.trim().toLowerCase();
    this.dataSource.data = [...this.dataSource.data];
  }

  showDetails(groupId: string): void {

    this.groupDetails.showDetails(groupId);

  }

  addBtnClick(schoolId: string) {
    const dialogRef = this.dialog.open(GroupFormComponent, {
      data: schoolId,
      disableClose: true
    });
  }

  editBtnClick(event: Event, group: IGroupModel, schoolId: string) {
    event.stopPropagation();
    const dialogRef = this.dialog.open(EditGroupFormComponent, {
      data: { group, schoolId },
      disableClose: true
    });
  }

}
