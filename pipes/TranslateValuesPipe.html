import { Pipe, PipeTransform } from '@angular/core';
import { Job, Role } from 'src/app/models/user/enum.model';
import { Province, Status, Type } from 'src/app/models/school/enum.model';
import { DaysOfTheWeek } from 'src/app/models/daysOfTheWeek.model'
import { Subject } from 'src/app/models/group/enum.model';

@Pipe({
  name: 'translateValues'
})
export class TranslateValuesPipe implements PipeTransform {

  transform(value: Role | Job | Type | Status | Province | DaysOfTheWeek | Subject): string {
    const roleLabels: { [key in Role]: string } = {
      [Role.MGTMT]: 'Zarząd',
      [Role.RD]: 'Dyrektor regionalny',
      [Role.CA]: 'Asystent centralny',
      [Role.USER]: 'Użytkownik'
    };
    const jobLabels: { [key in Job]: string } = {
      [Job.EDUCATIONMGR]: 'Menadżer edukacji',
      [Job.CAS]: 'CAS',
      [Job.ROBOTICSINSTRUCTOR]: 'Instruktor robotyki',
      [Job.PROGRAMINGINSTRUCTOR]: 'Instruktor programowania',
      [Job.LECTOR]: 'Lektor',
      [Job.FLYER]: 'Ulotkarz',
      [Job.OTHER]: 'Inne'
    };
    const typeLabels: { [key in Type]: string } = {
      [Type.PRK]: 'Przedszkole prywatne',
      [Type.PUK]: 'Przedszkole publiczne',
      [Type.PPS]: 'Szkoła podstawowa publiczna',
      [Type.PRPS]: 'Szkoła podstawowa prywatna',
      [Type.HS]: 'Szkoła srednia',
      [Type.O]: 'Inne'
    }
    const statusLabels: { [key in Status]: string } = {
      [Status.PD]: 'Po tel- brak zgody',
      [Status.PA]: 'Po tel - zgody',
      [Status.PND]: 'Po tel - brak decyzji',
      [Status.VD]: 'Po wizycie - brak zgody',
      [Status.VA]: 'Po wizycie - zgoda',
      [Status.VND]: 'Po wizycie - brak decyzji',
      [Status.NC]: 'Brak kontaktu z dyrekcją'
    }
    const provinceLabels: { [key in Province]: string } = {
      [Province.Dolnoslaskie]: 'Dolnoślaskie',
      [Province.KujawskoPomorskie]: 'Kujawsko Pomorskie',
      [Province.Lubelskie]: 'Lubelskie',
      [Province.Lubuskie]: 'Lubuskie',
      [Province.Lodzkie]: 'Łódzkie',
      [Province.Malopolskie]: 'Małopolskie',
      [Province.Mazowieckie]: 'Mazowieckie',
      [Province.Opolskie]: 'Opolskie',
      [Province.Podkarpackie]: 'Podkarpackie',
      [Province.Podlaskie]: 'Podlaskie',
      [Province.Pomorskie]: 'Pomorskie',
      [Province.Slaskie]: 'Ślaskie',
      [Province.Swietokrzyskie]: 'Świetokrzyskie',
      [Province.WarminskoMazurskie]: 'Warminsko Mazurskie',
      [Province.Wielkopolskie]: 'Wielkopolskie',
      [Province.ZachodnioPomorskie]: 'Zachodnio Pomorskie'
    }
    const daysOfWeekLabels: { [key in DaysOfTheWeek]: string } = {
      [DaysOfTheWeek.MONDAY]: 'Poniedziałek',
      [DaysOfTheWeek.TUESDAY]: 'Wtorek',
      [DaysOfTheWeek.WEDNESDAY]: 'Środa',
      [DaysOfTheWeek.THURSDAY]: 'Czwartek',
      [DaysOfTheWeek.FRIDAY]: 'Piątek',
      [DaysOfTheWeek.SATURDAY]: 'Sobota',
      [DaysOfTheWeek.SUNDAY]: 'Niedziela',
    }
    const subjectList: { [key in Subject]: string } = {
      [Subject.ROBOTICS]: 'Robotyka',
      [Subject.PROGGRAMING]: 'Programowanie',
    };
    if (Object.values(Role).includes(value as Role)) {
      return roleLabels[value as Role];
    }
    if (Object.values(Job).includes(value as Job)) {
      return jobLabels[value as Job];
    }
    if (Object.values(Type).includes(value as Type)) {
      return typeLabels[value as Type];
    }
    if (Object.values(Status).includes(value as Status)) {
      return statusLabels[value as Status];
    }
    if (Object.values(Province).includes(value as Province)) {
      return provinceLabels[value as Province];
    }
    if (Object.values(DaysOfTheWeek).includes(value as DaysOfTheWeek)) {
      return daysOfWeekLabels[value as DaysOfTheWeek];
    }
    if (Object.values(Subject).includes(value as Subject)) {
      return subjectList[value as Subject];
    }
    return value;
  }

}
