import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable, Subject, tap } from 'rxjs';
import { IFiszkaModel } from 'src/app/models/fiszka/fiszka.model';
import { IUpdateFiszkaModel } from 'src/app/models/fiszka/update-fiszka.model';
import { environment } from 'src/environments/environment';

@Injectable({
  providedIn: 'root'
})
export class FiszkaService {

  constructor(private http: HttpClient) { }

  fiszkiHasChanged = new Subject();
  private constrollerPath = 'fiszka'

  getFiszkiByRegion(): Observable<IFiszkaModel[]> {

    const url = `${environment.apiUrl}/${this.constrollerPath}/findallforregion`

    return this.http.get<IFiszkaModel[]>(url);

  }

  getFiszkiByInterviewId(interviewId: string): Observable<IFiszkaModel[]> {

    const url = `${environment.apiUrl}/${this.constrollerPath}/findall/${interviewId}`

    return this.http.get<IFiszkaModel[]>(url);

  }

  getFiszkiByGroupId(groupId: string): Observable<IFiszkaModel[]> {

    const url = `${environment.apiUrl}/${this.constrollerPath}/findallforgroup/${groupId}`

    return this.http.get<IFiszkaModel[]>(url);

  }

  addFiszkaToInterview(fiszka: IFiszkaModel, interviewId: string): Observable<IFiszkaModel> {

    const url = `${environment.apiUrl}/${this.constrollerPath}/addfiszki/${interviewId}`

    console.log('addFiszkaToInterview', fiszka, 'interviewId', interviewId);

    return this.http.post<IFiszkaModel>(url, fiszka).pipe(
      tap(() => this.fiszkiHasChanged.next(true))
    );

  }

  deleteFiszkaFromGroup(fiszkaId: string, groupId: string): Observable<IFiszkaModel> {

    const url = `${environment.apiUrl}/${this.constrollerPath}/deletefiszka/${fiszkaId}/fromgroup/${groupId}`

    console.log(`deleteFiszka ${fiszkaId}, groupdId ${groupId}`);

    return this.http.delete<IFiszkaModel>(url).pipe(
      tap(() => this.fiszkiHasChanged.next(true))
    );

  }

  addFiszkaToGroup(fiszka: IFiszkaModel, groupId: string): Observable<IFiszkaModel> {

    const url = `${environment.apiUrl}/${this.constrollerPath}/addgroup/${groupId}`

    console.log('addFiszkaToInterview', fiszka, 'groupId', groupId);

    return this.http.patch<IFiszkaModel>(url, fiszka).pipe(
      tap(() => this.fiszkiHasChanged.next(true))
    );

  }

  updateFiszka(updatedFiszka: IUpdateFiszkaModel, fiszkaId: string): Observable<IUpdateFiszkaModel> {

    const url = `${environment.apiUrl}/${this.constrollerPath}/editfiszka/${fiszkaId}`

    console.log('updateFiszka', updatedFiszka, 'interviewId');

    return this.http.patch<IUpdateFiszkaModel>(url, updatedFiszka).pipe(
      tap(() => this.fiszkiHasChanged.next(true))
    );

  }

  deleteFiszka(fiszkaId: string): Observable<IFiszkaModel> {

    const url = `${environment.apiUrl}/${this.constrollerPath}/deletefiszka/${fiszkaId}`

    return this.http.delete<IFiszkaModel>(url).pipe(
      tap(() => this.fiszkiHasChanged.next(true))
    );

  }

}
